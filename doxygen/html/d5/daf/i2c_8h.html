uint8_t i;

I2C.DR.byte = (uint8_t) ((slaveAddr << 1) | 0x01); // shift left and set LSB (write=0, read=1)
while ((!I2C.SR1.reg.ADDR) && (!check_timeout())); // wait until adress sent or timeout

// receive data from slave

bufRx[i] = I2C.DR.byte;
}
// on I2C timeout set error flag
if (check_timeout()) {
stop_timeout_ms();
return(ERROR_TIMOUT);
}
// reset timeout timer
stop_timeout_ms();
return(SUCCESS);
